-- Step 1: Drop the existing BOOKS table if it exists
DROP TABLE BOOKS CASCADE CONSTRAINTS;

-- Step 2: Create the SUPPLIERS table
CREATE TABLE SUPPLIERS (
    SUPPLIER_ID INT PRIMARY KEY,
    CONTACT VARCHAR2(100),
    PHONE VARCHAR2(15)
);

-- Step 3: Create the BOOKS table
CREATE TABLE BOOKS (
    BOOK_ID INT PRIMARY KEY,
    TITLE VARCHAR2(255),
    SUPPLIER_ID INT,
    FOREIGN KEY (SUPPLIER_ID) REFERENCES SUPPLIERS(SUPPLIER_ID)
);

-- Step 4: Create the REORDERINFO view
CREATE VIEW REORDERINFO AS
SELECT b.BOOK_ID, b.TITLE, s.CONTACT, s.PHONE
FROM BOOKS b
JOIN SUPPLIERS s ON b.SUPPLIER_ID = s.SUPPLIER_ID;

ROLLBACK;

DROP VIEW REORDERINFO;
